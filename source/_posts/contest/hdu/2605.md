```java
import java.io.IOException;
import java.io.InputStream;
import java.util.*;

public class Main {
    private static FastReader cin = new FastReader(System.in);

    private static final int N = 20 * 20;
    private static final int ST = 1 << 7;

    private static final int EMPTY = 0;
    private static final int WALL = 1;
    private static final int BEAN = 2;
    private static final int BODY = 3;
    private static final int UNKNOWN = 4;
    private static final int HAS_BEAN_CAN_NOT_EAT = 5;
    private static final int HAS_BEAN_AND_EAT = 6;

    private static int h, w, l, k, r, n;
    private static final int[] beanX = new int[10];
    private static final int[] beanY = new int[10];
    private static final int[] key = new int[N];
    private static final int[] next = new int[N];
    private static final int[] wallX = new int[N];
    private static final int[] wallY = new int[N];

    private static int[][][] dp = new int[21][21][ST];


    private static void init() {
        for (int i = 0; i < 21; ++i) {
            for (int j = 0; j < 21; ++j) {
                for (int st = 0; st < ST; ++st) {
                    dp[i][j][st] = Integer.MAX_VALUE;
                }
            }
        }
    }

    private static int[] dirX = new int[]{1, -1, 0, 0};
    private static int[] dirY = new int[]{0, 0, 1, -1};
    private static int[][] beanId = new int[21][21];

    private static String showSt(int st) {
        String res = "";
        for (int i = 0; i < 7; ++i) {
            if ((st & (1 << i)) == 0) res += "0";
            else res += "1";
        }
        return res;
    }

    private static void dfs(LinkedList<Integer> bodyX,
                            LinkedList<Integer> bodyY,
                            int st, int step, int[][] map) {
        int x = bodyX.getFirst();
        int y = bodyY.getFirst();
        if (dp[x][y][st] >= step) {

            dp[x][y][st] = step;
            if (step <= 21) System.out.println("x:" + x + ",y:" + y + ",st:" + showSt(st) + ",step:" + step);

            int lastX = bodyX.getLast();
            int lastY = bodyY.getLast();
            bodyX.removeLast();
            bodyY.removeLast();
            map[lastX][lastY] = EMPTY;

            for (int d = 0; d < 4; ++d) {
                int nx = x + dirX[d];
                int ny = y + dirY[d];
                int nSt = st;
                if (nx <= 0 || ny <= 0 || nx > h || ny > w) continue;

                int now = map[nx][ny];
                if (now == WALL || now == BODY) continue;

                int eat = UNKNOWN;

                if (now == BEAN) {
                    int id = beanId[nx][ny];
                    for (int i = 0; i < r; ++i) {
                        if (next[i] == id && (((1 << key[i]) & st) == 0)) {
                            eat = HAS_BEAN_CAN_NOT_EAT;
                            break;
                        }
                    }
                    if (eat == UNKNOWN) {
                        eat = HAS_BEAN_AND_EAT;
                        nSt = st | (1 << id);
                    }
                }

                map[nx][ny] = BODY;
                bodyX.addFirst(nx);
                bodyY.addFirst(ny);
                dfs(bodyX, bodyY, nSt, step + 1, map);
                // 恢复
                if (eat == HAS_BEAN_AND_EAT) map[nx][ny] = BEAN;
                else map[nx][ny] = EMPTY;
                bodyX.removeFirst();
                bodyY.removeFirst();
            }

            bodyX.addLast(lastX);
            bodyY.addLast(lastY);
            map[lastX][lastY] = BODY;
        }
    }

    private static void solve() {

        h = cin.nextInt();
        w = cin.nextInt();
        l = cin.nextInt();
        k = cin.nextInt();
        r = cin.nextInt();
        n = cin.nextInt();

        int[][] map = new int[21][21];

        LinkedList<Integer> bodyX = new LinkedList<>();
        LinkedList<Integer> bodyY = new LinkedList<>();

        for (int i = 0; i < l; ++i) {
            int x = cin.nextInt();
            int y = cin.nextInt();
            bodyX.add(x);
            bodyY.add(y);
            map[x][y] = BODY;
        }

        for (int i = 0; i < k; ++i) {
            beanX[i] = cin.nextInt();
            beanY[i] = cin.nextInt();
            map[beanX[i]][beanY[i]] = BEAN;
            beanId[beanX[i]][beanY[i]] = i;
        }

        for (int i = 0; i < r; ++i) {
            key[i] = cin.nextInt() - 1;
            next[i] = cin.nextInt() - 1;
        }

        for (int i = 0; i < n; ++i) {
            wallX[i] = cin.nextInt();
            wallY[i] = cin.nextInt();
            map[wallX[i]][wallY[i]] = WALL;
        }

        init();


        dfs(bodyX, bodyY, 0, 0, map);

        int ans = Integer.MAX_VALUE;
        for (int i = 1; i <= h; ++i) {
            for (int j = 1; j <= w; ++j) {
                ans = Math.min(ans, dp[i][j][(1 << k) - 1]);
            }
        }

        System.out.println(ans == Integer.MAX_VALUE ? -1 : ans);
    }


    public static void main(String[] args) {
        int t = cin.nextInt();
        for (int tt = 0; tt < t; ++tt) solve();
    }


    private static class FastReader {
        private InputStream stream;

        private byte[] buf = new byte[1024];

        private int curChar;

        private int numChars;

        FastReader(InputStream stream) {
            this.stream = stream;
        }

        int read() {
            if (numChars == -1) {
                throw new InputMismatchException();
            }
            if (curChar >= numChars) {
                curChar = 0;
                try {
                    numChars = stream.read(buf);
                } catch (IOException e) {
                    throw new InputMismatchException();
                }
                if (numChars <= 0) {
                    return -1;
                }
            }
            return buf[curChar++];
        }

        int nextInt() {
            int c = read();
            while (isSpaceChar(c)) {
                c = read();
            }
            int sgn = 1;
            if (c == '-') {
                sgn = -1;
                c = read();
            }
            int res = 0;
            do {
                if (c < '0' || c > '9') {
                    throw new InputMismatchException();
                }
                res *= 10;
                res += c - '0';
                c = read();
            } while (!isSpaceChar(c));
            return res * sgn;
        }

        public long nextLong() {
            long ret = 0;
            int c = read();
            while (c <= ' ') {
                c = read();
            }
            boolean neg = (c == '-');
            if (neg) {
                c = read();
            }
            do {
                ret = ret * 10 + c - '0';
            } while ((c = read()) >= '0' && c <= '9');
            if (neg) {
                return -ret;
            }
            return ret;
        }

        public String next() {
            int c = read();
            while (isSpaceChar(c)) {
                c = read();
            }
            StringBuilder res = new StringBuilder();
            do {
                res.appendCodePoint(c);
                c = read();
            } while (!isSpaceChar(c));
            return res.toString();
        }

        boolean isSpaceChar(int c) {
            return c == ' ' || c == '\n' || c == '\r' || c == '\t' || c == -1;
        }
    }
}
```
