import java.io.IOException;
import java.io.InputStream;
import java.util.*;

public class Main {
    private static FastReader cin = new FastReader(System.in);

    private static final int NN = 20 * 20;
    private static final int ST = 1 << 7;

    private static final int EMPTY = 0;
    private static final int WALL = 1;
    private static final int BEAN = 2;
    private static final int BODY = 3;
    private static final int UNKNOWN = 4;
    private static final int HAS_BEAN_CAN_NOT_EAT = 5;
    private static final int HAS_BEAN_AND_EAT = 6;

    private static int H, W, L, K, R, n;
    private static final int[] beanX = new int[10];
    private static final int[] beanY = new int[10];
    private static final int[] key = new int[NN];
    private static final int[] next = new int[NN];
    private static final int[] wallX = new int[NN];
    private static final int[] wallY = new int[NN];

    private static int[][][][] dp = new int[21][21][ST][8];


    private static void init() {
        for (int i = 0; i < 21; ++i) {
            for (int j = 0; j < 21; ++j) {
                for (int st = 0; st < ST; ++st) {
                    for (int k = 0; k < 8; ++k) {
                        dp[i][j][st][k] = Integer.MAX_VALUE;
                    }
                }
            }
        }
    }

    private static int[] dirX = new int[]{1, -1, 0, 0};
    private static int[] dirY = new int[]{0, 0, 1, -1};
    private static int[][] beanId = new int[21][21];

    private static void dfs(LinkedList<Integer> bodyX,
                            LinkedList<Integer> bodyY,
                            int st, int step, int[][] map) {

        if (step > H * W * 2) return;

        int x = bodyX.getFirst();
        int y = bodyY.getFirst();

        int k = -1;
        for (int i = 0; i < L; ++i) {
            if (dp[x][y][st][i] > step) {
                k = i;
                break;
            }
        }

        if (k != -1) {
            dp[x][y][st][k] = step;
//            if (step <= 21) {
//                System.out.println("x:" + x + ",y:" + y + ",st:" + showSt(st) + ",step:" + step);
//            }
            int lastX = bodyX.getLast();
            int lastY = bodyY.getLast();
            bodyX.removeLast();
            bodyY.removeLast();
            map[lastX][lastY] = EMPTY;

            for (int d = 0; d < 4; ++d) {
                int nx = x + dirX[d];
                int ny = y + dirY[d];
                int nSt = st;
                if (nx <= 0 || ny <= 0 || nx > H || ny > W) continue;

                int now = map[nx][ny];
                if (now == WALL || now == BODY) continue;

                int eat = UNKNOWN;

                if (now == BEAN) {
                    int id = beanId[nx][ny];
                    for (int i = 0; i < R; ++i) {
                        if (next[i] == id && (((1 << key[i]) & st) == 0)) {
                            eat = HAS_BEAN_CAN_NOT_EAT;
                            break;
                        }
                    }
                    if (eat == UNKNOWN) {
                        eat = HAS_BEAN_AND_EAT;
                        nSt = st | (1 << id);
                    }
                }

                if (eat == HAS_BEAN_CAN_NOT_EAT) continue;

                map[nx][ny] = BODY;
                bodyX.addFirst(nx);
                bodyY.addFirst(ny);
                dfs(bodyX, bodyY, nSt, step + 1, map);
                // 恢复
                map[nx][ny] = now;
                bodyX.removeFirst();
                bodyY.removeFirst();
            }

            bodyX.addLast(lastX);
            bodyY.addLast(lastY);
            map[lastX][lastY] = BODY;
        }
    }

    private static void solve() {

        H = cin.nextInt();
        W = cin.nextInt();
        L = cin.nextInt();
        K = cin.nextInt();
        R = cin.nextInt();
        n = cin.nextInt();

        int[][] map = new int[21][21];

        LinkedList<Integer> bodyX = new LinkedList<>();
        LinkedList<Integer> bodyY = new LinkedList<>();

        for (int i = 0; i < L; ++i) {
            int x = cin.nextInt();
            int y = cin.nextInt();
            bodyX.add(x);
            bodyY.add(y);
            map[x][y] = BODY;
        }

        for (int i = 0; i < K; ++i) {
            beanX[i] = cin.nextInt();
            beanY[i] = cin.nextInt();
            map[beanX[i]][beanY[i]] = BEAN;
            beanId[beanX[i]][beanY[i]] = i;
        }

        for (int i = 0; i < R; ++i) {
            key[i] = cin.nextInt() - 1;
            next[i] = cin.nextInt() - 1;
        }

        for (int i = 0; i < n; ++i) {
            wallX[i] = cin.nextInt();
            wallY[i] = cin.nextInt();
            map[wallX[i]][wallY[i]] = WALL;
        }

        init();


        dfs(bodyX, bodyY, 0, 0, map);

        int ans = Integer.MAX_VALUE;
        for (int i = 1; i <= H; ++i) {
            for (int j = 1; j <= W; ++j) {
                for (int k = 0; k < 8; ++k) {
                    ans = Math.min(ans, dp[i][j][(1 << K) - 1][k]);
                }
            }
        }

        System.out.println(ans == Integer.MAX_VALUE ? -1 : ans);
    }


    public static void main(String[] args) {
        int t = cin.nextInt();
        for (int tt = 0; tt < t; ++tt) solve();
    }

    private static String showSt(int st) {
        String res = "";
        for (int i = 0; i < 7; ++i) {
            if ((st & (1 << i)) == 0) res += "0";
            else res += "1";
        }
        return res;
    }

    private static class FastReader {
        private InputStream stream;

        private byte[] buf = new byte[1024];

        private int curChar;

        private int numChars;

        FastReader(InputStream stream) {
            this.stream = stream;
        }

        int read() {
            if (numChars == -1) {
                throw new InputMismatchException();
            }
            if (curChar >= numChars) {
                curChar = 0;
                try {
                    numChars = stream.read(buf);
                } catch (IOException e) {
                    throw new InputMismatchException();
                }
                if (numChars <= 0) {
                    return -1;
                }
            }
            return buf[curChar++];
        }

        int nextInt() {
            int c = read();
            while (isSpaceChar(c)) {
                c = read();
            }
            int sgn = 1;
            if (c == '-') {
                sgn = -1;
                c = read();
            }
            int res = 0;
            do {
                if (c < '0' || c > '9') {
                    throw new InputMismatchException();
                }
                res *= 10;
                res += c - '0';
                c = read();
            } while (!isSpaceChar(c));
            return res * sgn;
        }

        public long nextLong() {
            long ret = 0;
            int c = read();
            while (c <= ' ') {
                c = read();
            }
            boolean neg = (c == '-');
            if (neg) {
                c = read();
            }
            do {
                ret = ret * 10 + c - '0';
            } while ((c = read()) >= '0' && c <= '9');
            if (neg) {
                return -ret;
            }
            return ret;
        }

        public String next() {
            int c = read();
            while (isSpaceChar(c)) {
                c = read();
            }
            StringBuilder res = new StringBuilder();
            do {
                res.appendCodePoint(c);
                c = read();
            } while (!isSpaceChar(c));
            return res.toString();
        }

        boolean isSpaceChar(int c) {
            return c == ' ' || c == '\n' || c == '\r' || c == '\t' || c == -1;
        }
    }
}

import java.io.IOException;
import java.io.InputStream;
import java.util.*;

public class Main {
    private static FastReader cin = new FastReader(System.in);

    private static final int NN = 20 * 20;
    private static final int ST = 1 << 7;

    private static final int EMPTY = 0;
    private static final int WALL = 1;
    private static final int BEAN = 2;
    private static final int BODY = 3;
    private static final int UNKNOWN = 4;
    private static final int HAS_BEAN_CAN_NOT_EAT = 5;
    private static final int HAS_BEAN_AND_EAT = 6;

    private static int H, W, L, K, R, n;
    private static final int[] beanX = new int[10];
    private static final int[] beanY = new int[10];
    private static final int[] key = new int[NN];
    private static final int[] next = new int[NN];
    private static final int[] wallX = new int[NN];
    private static final int[] wallY = new int[NN];

    private static final int[][][] dp = new int[21][21][ST];

    private static void init() {
        for (int i = 0; i <= H; ++i) {
            for (int j = 0; j <= W; ++j) {
                for (int st = 0; st < ST; ++st) {
                    dp[i][j][st] = Integer.MAX_VALUE;
                }
            }
        }
    }

    private static final int[] dirX = new int[]{1, 0, 0, -1};
    private static final int[] dirY = new int[]{0, 1, -1, 0};
    private static final int[][] beanId = new int[21][21];

    private static int encodePos(int x, int y) {
        return x * W + y;
    }

    private static int getX(int code) {
        return code / W;
    }

    private static int getY(int code) {
        return code % W;
    }

    private static int encodeBody(List<Integer> bodyX, List<Integer> bodyY) {
        int[] target = new int[L];
        for (int i = 0; i < L - 1; ++i) {
            int x = bodyX.get(i);
            int y = bodyY.get(i);
            int nxtX = bodyX.get(i + 1);
            int nxtY = bodyY.get(i + 1);
            for (int d = 0; d < 4; ++d) {
                if (x + dirX[d] == nxtX && y + dirY[d] == nxtY) {
                    target[i] = d;
                    break;
                }
            }
        }
        int res = 0;
        for (int i = L - 2; i >= 0; --i) {
            res *= 4;
            res += target[i];
        }
        return res;
    }

    private static List<Integer> decode(int begin, int code, int[] dir) {
        List<Integer> res = new ArrayList<>();
        res.add(begin);
        int pre = begin;
        for (int i = 1; i < L; ++i) {
            int d = code % 4;
            code /= 4;
            int now = pre + dir[d];
            res.add(now);
            pre = now;
        }
        return res;
    }

    private static int move(int body, int d) {
        int nBody = body * 4 + (3 - d);
        int st = 1 << (L * 2);
        return nBody & (st - 1);
    }

    private static boolean isBody(int headX, int headY, int x, int y, int code) {
        if (x == headX && y == headY) return true;
        int nowX = headX;
        int nowY = headY;
        for (int i = 1; i < L; ++i) {
            int d = code % 4;
            code /= 4;
            nowX += dirX[d];
            nowY += dirY[d];
            if (x == nowX && y == nowY) return true;
        }
        return false;
    }

    private static void checkEncode(List<Integer> bodyX, List<Integer> bodyY) {
        int code = encodeBody(bodyX, bodyY);
        List<Integer> decodeX = decode(bodyX.get(0), code, dirX);
        List<Integer> decodeY = decode(bodyY.get(0), code, dirY);

        for (int i = 0; i < L; ++i) {
            System.out.println(bodyX.get(i) + " " + bodyY.get(i) + " " + decodeX.get(i) + " " + decodeY.get(i));

        }
    }

    public static final class Node {
        int x;
        int y;
        int st;
        int step;
        int body;

        Node(int x, int y, int st, int step, int body) {

            this.step = step;

        }

    private static int bfs(int x, int y, int body, int[][] map) {
        Queue<Node> queue = new LinkedList<>();
        queue.add(new Node(x, y, 0, 0, body));
        dp[x][y][0] = 0;
        while (!queue.isEmpty()) {
            Node top = queue.poll();
            if (top.st == (1 << K) - 1) return top.step;
            if (dp[top.x][top.y][top.st] != Integer.MAX_VALUE && top.step > dp[top.x][top.y][top.st] + L) continue;
//            top.show();
            // if (top.step > dp[top.x][top.y][top.st] + L * 3) continue;
            for (int d = 0; d < 4; ++d) {
                int nx = top.x + dirX[d];
                int ny = top.y + dirY[d];
                int nSt = top.st;
                if (nx <= 0 || ny <= 0 || nx > H || ny > W) continue;

                int now = map[nx][ny];
                if (now == WALL) continue;
                if (isBody(top.x, top.y, nx, ny, top.body)) continue;

                int eat = UNKNOWN;
                if (now == BEAN) {
                    int id = beanId[nx][ny];
                    for (int i = 0; i < R; ++i) {
                        if (next[i] == id && (((1 << key[i]) & top.st) == 0)) {
                            eat = HAS_BEAN_CAN_NOT_EAT;
                            break;
                        }
                    }
                    if (eat == UNKNOWN) {
                        eat = HAS_BEAN_AND_EAT;
                        nSt = top.st | (1 << id);
                    }
                }

                if (eat == HAS_BEAN_CAN_NOT_EAT) continue;
                if (dp[nx][ny][nSt] != Integer.MAX_VALUE || top.step + 1 < dp[nx][ny][nSt]) {
                    dp[nx][ny][nSt] = top.step + 1;
                }
                int nBody = move(top.body, d);
                queue.add(new Node(nx, ny, nSt, top.step + 1, nBody));
            }
        }
        return Integer.MAX_VALUE;
    }

    private static boolean checkCircle() {
        return false;
    }

        H = cin.nextInt();

        L = cin.nextInt();

            wallX[i] = cin.nextInt();
            wallY[i] = cin.nextInt();
            map[wallX[i]][wallY[i]] = WALL;
        }

        }

        for (int i = 0; i < L; ++i) System.out.print(bodyX.get(i) + " " + bodyY.get(i) + ",");
        System.out.println();
        int nx = bodyX.get(0) + dirX[1];

    }
